    //input
    for(int i=0;i<=n-1;i++){
        cin>>arr[i];
    }
    int i=0;
    int j=n-1;
    bool flag=true;
    while(i<j){
        if(arr[i] != arr[j]){
            flag=false;
            break;
        }
        i++;
        j--;
    }
    if(flag==true){
        cout<<"array is palindrome";
    }
    else{
        cout<<"array is not palindrome";
    }
}
